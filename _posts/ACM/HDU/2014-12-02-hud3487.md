###题意



>给定一个长度为n(n <= 3*10^5)的数组，有两种类型的操作**:

>1.将区间[l,r]内的数剪掉，整体添加到新数组的c位置之后

>2.将区间[l,r]内的数反转,例如：1,2,3变成3,2,1

>总操作数为m(m <= 3*10^5)，在所有操作后，输出这个数组，时间限制2s。



###分析



区间反转，删除是Splay的典型应用啊，这题也是入门题。

>对于操作1：将l-1 splay到根，r+1 splay到根的右子树，这样r+1的左子树就是区间[l,r]，然后将r+1的左子树删除得到的该区间，再将c splay到根，将c+1 splay到根的右子树，那么此时根右子树的左子树必然数空树，再将区间刚才删除的[l,r]接到跟的右子树的左子树即可。

>对于操作2: 将区间[l,r] splay到根右子树的左子树上，操作和1的spaly一样，然后打上延迟标记，就像线段树的额延迟标记一样，这里的延迟标记表示的是以某个节点为根的子树全体需要进行反转操作。

#pragma comment(linker, "/STACK:1024000000,1024000000")

#include<iostream>
#include<algorithm>
#define sc scanf
#define eps 1e-6
#define fi first
#define se second
#define dl double
#define pr printf
using namespace std;
#define pb push_back
#define mp make_pair
#define ll long long
#define inf 0x7fffffff
#define vi vector<int>
#define vii vector<VI>
#define ls(k) ((k)<<1)
#define rs(k) (ls(k)|1)
const dl pi = acos(-1.0);
#define mii map<int,int>
#define pii pair<int,int>
#define vs vector<string>
#define mis map<int,string>
#define msi map<string,int>
#define debug(x) cout<<x<<endl
#define half(a,b) (((a)+(b))>>1)
#define all(c) c.begin(),c.end()
#define fillc(c,v) fill(all(c),v)
#define len(a) sizeof(a)/sizeof(a[0])
#define filla(a,b) fill(a,a+len(a),b)
#define clr(a,b) memset(a,b,sizeof(a))
#define forn(i,n) for(int i=0;i<n;i++)
#define out(file) freopen(file,"w",stdout)
#define input(file) freopen(file,"r",stdin)
#define fori(i,from,to) for(int i=from;i<to;i++)
#define ford(i,from,to) for(int i=from;i>to;i--)
#define forc(it,v) for(typeof(v.begin())it=v.begin();it!=v.end();it++)
#define key_Value ch[ch[root][1]][0]
const int MAXN = 3e5+10;
int size[MAXN],ch[MAXN][2],pre[MAXN];
int key[MAXN],lazy[MAXN],cnt,root,tot,n;
//向上更新，把孩子的信息推上来;
inline void pushUp(int r){
    size[r] = size[ch[r][0]] + size[ch[r][1]] + 1;
}
//更新某个节点的lazy值;
inline void updateLazy(int r){
    if(!r) return;
    swap(ch[r][0],ch[r][1]);
    lazy[r] ^= 1;
}
//向下更新，把父节点的lazy传下来;
inline void pushDown(int r){
    if(lazy[r]){
        updateLazy(ch[r][0]);
        updateLazy(ch[r][1]);
        lazy[r] = 0;
    }
}
//创建一个新节点;
inline void newNode(int &r,int fa,int k){
    r = ++tot;
    key[r] = k,pre[r] = fa;
    ch[r][0] = ch[r][1] = 0;
    size[r] = 1,lazy[r] = 0;
}
//建树，这里是按下标从数组中间向两边建;
void build(int &x,int l,int r,int fa){
    if(l > r) return;
    int mid = (l + r) >> 1;
    newNode(x,fa,mid);
    build(ch[x][0],l,mid - 1,x);
    build(ch[x][1],mid + 1,r,x);
    pushUp(x);
}
//初始化;
void init(){
    root = tot = 0;
    ch[0][0] = ch[0][1] = size[0] = 0;
    key[0] = pre[0] = lazy[0] = 0;
    //建两个需节点(0,n+1),方便找区间时统一处理;
    newNode(root,0,-1);
    newNode(ch[root][1],root,-1);
    //建树;
    build(key_Value,1,n,ch[root][1]);
    //把孩子的值推上来;
    pushUp(ch[root][1]);
    pushUp(root);
}
//旋转操作，kind=0,左旋kind=1,右旋;
void rotate(int x,int kind){
    int y = pre[x];
    //因为要旋转y,x,所以要把父y,x节点的lazy值传下去;
    pushDown(y),pushDown(x);
    //把x的孩子挂到y的下面;
    ch[y][!kind] = ch[x][kind];
    pre[ch[x][kind]] = y;
    //把x挂到y的父亲上;
    if(pre[y]) ch[pre[y]][ch[pre[y]][1] == y] = x;
    pre[x] = pre[y];
    //把y挂到x下面;
    ch[x][kind] = y;
    pre[y] = x;
    //因为y旋转到了x的下面，要更新y节点的值;
    pushUp(y);
}
//关键点，伸展操作，作用是把r旋转到goal下面;
void splay(int r,int goal){
    //把r的信息传下去，因为x会被旋转上去;
    pushDown(r);
    while(pre[r] != goal){
        //父节点的父节点是goal,直接单旋;
        if(pre[pre[r]] == goal){
            rotate(r,ch[pre[r]][0] == r);
        }else{
            int y = pre[r],kind = ch[pre[y]][0] == y;
            if(ch[y][kind] == r){
                //之字型旋转;
                rotate(r,!kind);
                rotate(r,kind);
            }else{
                //一字型旋转;
                rotate(y,kind);
                rotate(r,kind);
            }
        }
    }
    //旋转后更新r的信息;
    pushUp(r);
    if(goal == 0) root = r;
}
//查找第k个元素;
int getKth(int r,int k){
    pushDown(r);
    int t = size[ch[r][0]]+1;
    if(t == k) return r;
    if(t > k) return getKth(ch[r][0],k);
    else return getKth(ch[r][1],k - t);
}
void cut(int l,int r,int c){
    splay(getKth(root,l),0);
    splay(getKth(root,r+2),root);
    int tmp = key_Value;
    key_Value = 0;
    pushUp(ch[root][1]),pushUp(root);
    splay(getKth(root,c+1),0);
    splay(getKth(root,c+2),root);
    key_Value = tmp;
    pre[key_Value] = ch[root][1];
    pushUp(ch[root][1]),pushUp(root);
}
void flip(int l,int r){
    splay(getKth(root,l),0); 
    splay(getKth(root,r+2),root);
    updateLazy(key_Value);
    pushUp(ch[root][1]),pushUp(root);
}
void output(int r){
    if(!r) return;
    pushDown(r);
    output(ch[r][0]);
    if(cnt >= 1 && cnt <= n){
        if(cnt < n) pr("%d ",key[r]);
        else pr("%d\n",key[r]);
    }
    cnt ++;
    output(ch[r][1]);
}
int main(){
    int m,a,b,c;
    char str[10];
    /* input("in.cpp"); */
    while(sc("%d%d",&n,&m) && n != -1 && m != -1){
        init();
        while(m--){
            sc("%s",str);
            if(str[0] == 'C'){
                sc("%d%d%d",&a,&b,&c);
                cut(a,b,c);
            }else{
                sc("%d%d",&a,&b);
                flip(a,b);
            }
        }
        cnt = 0;
        output(root);
    }
    return 0;
}
